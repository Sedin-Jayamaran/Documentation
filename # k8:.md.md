#### Replicaset :
```
apiVersion: apps/v1
kind: ReplicaSet
metadata:
    name: myapp-replicaset
    labels:
        apps: myapp
spec:
    selector:
        matchLabels:
            env: production -------> This must match the label of pod which we are about to replicate
    replicas: 3
    template:
        metadata:
            name: myapp-pod
            labels:
                type: front-end
        spec:   
            containers:
                - name: nginx
                  image: nginx
```
What important is to match the labels of selectors and labels inside the pods . To view the detials about replicaset we need to use ``` kubectl describe replicaset myapp-replica```

#### Deployment.yml
```
apiVersion: apps/v1
kind: Deployment
metadata:
    name: myapp-deployment
    labels:
        tier: front-end
        app: nginx
spec:
    selector:
        matchLabels:
            app: myapp
    replicas: 3
    template:
        metadata:
            labels:
                type: front-end
                app: myapp
        spec:
            containers:
                - name: nginx
                  image: nginx
```
